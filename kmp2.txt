void get_next( int *next,char *a,int la) /*求NEXT[]的值*/
{ 
   int i=1,j=0 ; 
   next[1] = 0 ; 
     
   while ( i <= la) /*核心部分*/
   { 
      if( a[i] == a[j] || j == 0 ) 
      { 
        j ++ ; 
        i ++ ; 
        if( a[i] == a[j]) 
        next[i] = next[j]; 
        else
        next[i] = j ; 
      } 
      else
      j = next[j] ; 
   } 
}
